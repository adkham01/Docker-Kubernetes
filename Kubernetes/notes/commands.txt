CREATING RESOURCES (IMPERATIVE)
===============================
# Pods
k run nginx --image nginx
k run redis --image=redis -n-finance
k run tmp-shell --rm -it --image bretfisher/netshoot -- bash

# Deployments
k create deployment my-nginx --image nginx
k create deployment my-apache --image httpd
k create deployment httpenv --image=bretfisher/httpenv
k create deployment test --image nginx --dry-run=client -o yaml

# Scaling
k scale deploy my-apache --replicas 2
k scale deployment httpenv --replicas=5
k scale deployment nginx --replicas=5
k scale replicaset myapp-replicaset --replicas=2

k expose deployment httpenv --port 8888
k expose deployment httpenv --port 8888 --name httpenv-np --type NodePort
k expose deployment/httpenv --port 8888 --name httpenv-lb --type LoadBalancer
k expose deployment java-app --port=8080 --target-port=8080

# Jobs
k create job test --image nginx --dry-run=client -o yaml

# Namespaces

# Service Accounts

OPENSSL (CERTIFICATES)
======================
openssl genrsa -out ca.key 2048
openssl req -new -key ca.key -subj "/CN=KUBERNETES-CA" -out ca.csr
openssl x509 -req -in ca.csr -signkey ca.key -out ca.crt
o x509 -in ca.crt -text -noout
openssl genrsa -out admin.key 2048
openssl req -new -key admin.key -subj "/CN=kube-admin" -out admin.csr
openssl x509 -req -in admin.csr -CA ca.crt -CAkey ca.key -out admin.crt
openssl req -new -key apiserver.key -subj "/CN=kube-apiserver" -out apiserver.csr -config openssl.cnf
openssl x509 -req -in apiserver.csr -CA ../ca.crt -CAkey ../ca.key -CAcreateserial -out apiserver.crt -extensions v3_req -extfile openssl.cnf -days 1000
